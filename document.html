JavaScript

You have plugin.js file in your js folder When you downloaded this plugin but if you didnt Download plugin file you can copy below code to a js file and set this name for example "plugin.js" now you can use this plugin in your code by <script> tag alike : <script src="js/plugin.js"></script>

(function($){
$.fn.slider = function(useroption){
    var option = $.extend({
        // default argument
        img : ["image/slider1.jpg","image/slider2.jpg","image/slider3.jpg"],
        arr : "image/arrow.png",
        arrowEnable : true,
        timerEnable : true,
        time : 2000
    },useroption)
    this.each(function (){
        let i = 0;
        var header = $(this);
        var slider = $(".slider");
        var arrowRight = $("").attr("src",option.arr).addClass("arrowRight");
        var arrowLeft = $("").attr("src",option.arr).addClass("arrowLeft");
        var img = $("").attr("src",option.img[i]).addClass("sliderimg");
        var buttonPack = [];
        slider.append(img);
        if(option.arrowEnable === true){
            slider.append(arrowRight).append(arrowLeft);
        }
        header.append(slider);
        if(option.arrowEnable === true){
            arrowRight.click(function (){
                i++;
                var p = i;
                if(i === option.img.length){
                    i = 0;
                }
                img.attr("src",option.img[i]);
                slider.append(img).append(arrowRight).append(arrowLeft);
                header.append(slider);
            });
            arrowLeft.click(function (){
                i--;
                var p = i;
                if(i === -1){
                    i = option.img.length -1;
                }
                img.attr("src",option.img[i]);
                slider.append(img).append(arrowRight).append(arrowLeft);
                header.append(slider);
            });
        }
        if(option.timerEnable === true){
            setInterval(function (){
                i++;
                var p =i;
                if(i === option.img.length){
                    i = 0;
                }
                img.attr("src",option.img[i]);
                slider.append(img);
                if(option.arrowEnable === true){
                    slider.append(arrowRight).append(arrowLeft);
                }
                header.append(slider);
            },option.time);
        }
    });
}
})(jQuery)

HTML

I ) Now you can added js and css file to your html code. you Just use the following two tags in your code
1 : <link rel="stylesheet" href="css/pluginStyle.css"/>
2 : <script src="js/plugin.js"></script>

<html>
    <head>
        <link rel="stylesheet" href="css/pluginStyle.css">
        <script src="js/plugin.js"></script>
    </head>
    <body>
    </body>
</html>

II ) Now you shoud adding two tags for using this plugin.

<html>
    <head>
        <link rel="stylesheet" href="css/pluginStyle.css">
        <script src="js/plugin.js"></script>
    </head>
    <body>
        <div class="header">
            <div class="slider"></div>
        </div>
    </body>
</html>

III ) And finally called plugin in your parent of slider tag ( to this example called on header )

 $(".header").slider({
    img : ["image/slider2.jpg","image/slider1.jpg","image/slider6.jpg","image/slider7.jpg"],
    arr : "image/arrow.png",
    arrowEnable: true,
    timerEnable: false,
    time : 1000
});

img : pass your image do you want to using in slider

arr : pass your arrow image do you want to using in slider (Next and Prev button)

arrowEnable : if you like use arrow to slider pass "treu" else pass "false" this value was "true" By default

timerEnable : If you like to change the image at one time a head pass "true" and else pass "false" this value was "true" By default

time : set a time to whenever changed images (ms) alike : (1000ms) = (1s)
